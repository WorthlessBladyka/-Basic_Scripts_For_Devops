## CREATE FOREIGN KEY
```sql
ALTER TABLE tabletest1 
ADD CONSTRAINT fk_tabletest1_tabletest2 
FOREIGN KEY (testLink) REFERENCES tabletest2(id);
```
## Сопоставление данных
```sql
-- Создать временную последовательность для сопоставления
WITH numbered_tabletest1 AS (
    SELECT id as t1_id, ROW_NUMBER() OVER () as rn
    FROM tabletest1
),
numbered_tabletest2 AS (
    SELECT id as t2_id, ROW_NUMBER() OVER () as rn
    FROM tabletest2
)
UPDATE tabletest1 
SET testLink = numbered_tabletest2.t2_id
FROM numbered_tabletest1, numbered_tabletest2
WHERE tabletest1.id = numbered_tabletest1.t1_id
AND numbered_tabletest1.rn = numbered_tabletest2.rn;
```
